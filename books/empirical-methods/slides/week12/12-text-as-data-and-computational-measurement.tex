\begin{frame}{Week 12 question}
\begin{itemize}
    \item How can raw text, images, audio, video, or multimodal records become economic variables?
    \item What latent object is being measured?
    \item Why does the raw modality contain signal about that object?
    \item How do measurement errors affect the descriptive, causal, or structural claim?
\end{itemize}

\medskip
The lecture is about unstructured data as a measurement environment for applied economics.
\end{frame}

\begin{frame}{Measurement model}
\small
\[
M_i = g(U_i;\theta)+\varepsilon_i
\]

\begin{tabular}{p{0.20\linewidth} p{0.70\linewidth}}
\toprule
Object & Applied interpretation \\
\midrule
\(U_i\) & raw document, image, audio clip, video, scan, or multimodal record \\
\(g(\cdot)\) & dictionary, classifier, embedding, prompt, or multimodal extractor \\
\(\theta\) & terms, labels, weights, thresholds, model parameters, prompts \\
\(M_i\) & remote-work score, exposure, risk, poverty proxy, tone, policy measure \\
\(\varepsilon_i\) & extraction error relative to the intended economic object \\
\bottomrule
\end{tabular}

\medskip
Validation is part of the design because the measure enters later economics.
\end{frame}

\begin{frame}{Dictionary and embedding measures}
\small
\[
D_i(S)=\frac{1}{N_i}\sum_{t\in U_i}\1\{t\in S\}
\]

\[
e_i=\frac{1}{N_i}\sum_{t\in U_i}h(t),
\qquad
S_i(c)=\frac{e_i'c}{\|e_i\|\|c\|}
\]

\begin{itemize}
    \item Dictionary shares are transparent and easy to audit.
    \item Embeddings broaden beyond exact terms to semantic similarity.
    \item Both require an economic interpretation before they become variables.
    \item The same score can mean mention, salience, exposure, or similarity depending on the design.
\end{itemize}
\end{frame}

\begin{frame}{Image and satellite applications}
\small
\begin{itemize}
    \item Satellite imagery can extend measurement to places with sparse surveys.
    \item Street-view imagery can measure built environment, amenities, storefronts, roads, or visible disorder.
    \item Scanned documents can turn paper archives into structured research inputs.
    \item Image measures need ground truth: surveys, administrative records, human labels, or known aggregates.
\end{itemize}

\medskip
Common failure modes:
\begin{itemize}
    \item sensor, season, weather, and resolution differences;
    \item proxy instability across countries or cities;
    \item visually salient features that are not welfare-relevant;
    \item domain shift between training and target places.
\end{itemize}
\end{frame}

\begin{frame}{Modality-specific failure modes}
\small
\begin{tabular}{p{0.18\linewidth} p{0.36\linewidth} p{0.34\linewidth}}
\toprule

\midrule

\medskip
The failure mode determines which downstream estimates are most fragile.
\end{frame}

\begin{frame}{Measurement error propagation}
\small
\[
\widehat M_i=M_i^*+\eta_i,
\qquad
Y_i=\alpha+\beta M_i^*+\nu_i
\]

\[
\plim \widehat\beta_{\widehat M}
=
\beta
\cdot
\frac{\operatorname{Var}(M_i^*)}
{\operatorname{Var}(M_i^*)+\operatorname{Var}(\eta_i)}
\quad
\text{under classical error}
\]

\begin{itemize}
    \item Classical noise attenuates simple regressions.
    \item Nonclassical errors can bias estimates in either direction.
    \item Errors often vary by occupation, firm, region, language, speaker, platform, or sensor.
    \item If the measure supplies the identifying variation, validation is part of identification.
\end{itemize}
\end{frame}

\begin{frame}{Takeaways}
\begin{itemize}
    \item Start with the economic object, not the modality or algorithm.
    \item Choose dictionaries, classifiers, embeddings, or multimodal features based on the mapping problem.
    \item Validate the constructed measure where it will be used.
    \item Report failure modes by time, group, geography, platform, and modality.
    \item Treat measurement error as a threat to downstream economics, not only as a prediction metric.
\end{itemize}
\end{frame}
